Cache:

    Instalación:

        apk add bind

    Configuración:

        * /etc/bind/named.conf.options

            acl whitelist {
                192.168.0.0/24;
                localhost;
                localnets;
                any;
            };

            options {
                directory "/var/cache/bind";
                version none;
                recursion yes;
                allow-query { whitelist; };
                dnssec-validation auto;
                auth-nxdomain no;
                listen-on-v6 { any; };
            };

Forward:

    Instalación:

        apk add bind

    Configuración:

        * /etc/bind/named.conf.options

            acl whitelist {
                192.168.0.0/24;
                localhost;
                localnets;
                any;
            };

            options {
                directory "/var/cache/bind";
                version none;
                recursion yes;
                allow-query { whitelist; };
                forward only;
                forwarders { 80.80.80.80; 80.80.81.81; 8.8.8.8; 8.8.4.4 };
                dnssec-enable yes;
                dnssec-validation yes;
                auth-nxdomain no;
                listen-on-v6 { any; };
            };

Authorative

    Instalación:

        apk add bind

    Configuración:

        Master:

            echo "<IP del servidor principal>    ns1"

            * /etc/bind/named.conf

                include "/etc/bind/named.conf.options";
                include "/etc/bind/named.conf.local";
                include "/etc/bind/named.conf.default-zones";

            * /etc/bind/named.conf.options

                options {
                    directory "/var/cache/bind";
                    recursion no;
                    allow-transfer { none; };
                    auth-nxdomain no;
                    listen-on-v6 { any; };
                    version none;
                };

            * /etc/bind/named.conf.local

                zone "<dominio>" {
                    type master;
                    file "/etc/bind/db.<dominio>";
                };

                zone "0.168.192.in-addr.arpa" {
                    type master;
                    file "/etc/bind/db.<dominio>.inv";
                };

                include "/etc/bind/zones.rfc1918";

            * /etc/bind/db.<dominio>

                $TTL    3600
                @       IN      SOA     <dominio>. root.<dominio>. (
                                   YYYYMMDDVV           ; Serial
                                         3600           ; Refresh [1h]
                                          600           ; Retry   [10m]
                                        86400           ; Expire  [1d]
                                          600 )         ; Negative Cache TTL [1h]

                @       IN      NS      ns.<dominio>.
                @       IN      MX      <prioridad> <dominio>.

                @       IN      A       <IP del servidor principal>
                ns      IN      A       <IP del servidor DNS>

                mail    IN      CNAME   ns

            * /etc/bind/db.<dominio>.inv

                $TTL    3600
                @ IN SOA        <dominio>. root.<dominio>. (
                                   YYYYMMDDVV           ; Serial
                                         3600           ; Refresh [1h]
                                          600           ; Retry   [10m]
                                        86400           ; Expire  [1d]
                                          600 )         ; Negative Cache TTL [1h]
                ;
                @       IN      NS      ns.<dominio>.

                <último octeto>    IN      PTR     <subdominio>.<dominio>.

            * /etc/resolv.conf

                search <dominio>

named-checkconf
named-checkconf -z
named-checkzone <dominio> /var/lib/bind/db.<dominio>
named-checkzone 0.168.192.in-addr.arpa /var/lib/bind/db.<dominio>.inv
