1. Crear el punto de montaje:
	$ mkdir /media/<nombreCarpeta> && chmod -R a=,u=rwx <nombreCarpeta>

2. Establecer el montaje automatico:
	# echo -ne "/dev/<unidad><partición>\t<puntoMontaje>\t<sistemaArchivos>\t<opcionesMontaje>\t<opcionesRespaldo>\t<opcionesVerificación>" >> /etc/fstab

	Ejemplo:
		# echo -ne "\n/dev/sdb3\t/media/Datos/\tntfs\tuser,rw,suid,dev,exec,auto\t0\t0" >> /etc/fstab

3. Reiniciar
	# reboot
